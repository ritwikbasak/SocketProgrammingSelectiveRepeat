Client Side -----

int sock = socket(PF_INET, SOCK_STREAM, IPPROTO_TCP);
connect(sock, (struct sockaddr*) &serverAddr, sizeof(serverAddr));
send (sock, msg, strlen(msg), 0);
recv (sock, recvBuffer, BUFSIZE - 1, 0); recvBuffer[bytesRecvd] = '\0'; //blocking call

Server Side -----

int serverSocket = socket (PF_INET, SOCK_STREAM, IPPROTO_TCP);
bind(serverSocket, (struct sockaddr*) &serverAddress, sizeof(serverAddress));
listen(serverSocket, MAXPENDING);
int clientSocket = accept (serverSocket, (struct sockaddr*) &clientAddress, &clientLength); //blocking call
recv(clientSocket, msg, BUFFERSIZE, 0); //blocking call
send (clientSocket,msg,strlen(msg),0);
close (serverSocket);
close (clientSocket);
